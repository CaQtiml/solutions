#include <stdio.h>

int main()
{
    int cnt[42] = {};
    for (int i = 0; i < 10; i++) {
        int num;
        scanf("%d", &num);
        cnt[num%42] = 1;
    }

    int ans = 0;
    for (int i = 0; i < 42; i++) {
        ans += cnt[i];
    }

    printf("%d", ans);
    return 0;
}
